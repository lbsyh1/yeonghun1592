```

  display: flex;
  flex-direction: column;
  text-align: right;
`;
const Section = styled.div`
  border: 1px solid black;
  background: #ffe9cf;
`;

const DisplayFlex = styled.div`
  display: flex;
`;
export default function () {
  return (
    <div>
      <Wrapper>
        <SearchBar />
      </Wrapper>
      <DisplayFlex>
        <Section>
          <Box foodName="" calories="" />
        </Section>
        <Section>
          <Box foodName="" calories="" />
        </Section>
      </DisplayFlex>
      <DisplayFlex>
        <Section>
          <Box foodName="" calories="" />
        </Section>
        <Section>
          <Box foodName="" calories="" />
        </Section>
      </DisplayFlex>
      <DisplayFlex>
        <Section>
          <Box foodName="" calories="" />
        </Section>
        <Section>
          <Box foodName="" calories="" />
        </Section>
      </DisplayFlex>
    </div>
    
    ```
